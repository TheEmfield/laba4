#!/bin/bash
if [[ $1 =~ ^[2-9][0-9]*$ ]] && [[ $# == 1 ]]
then
for(( i=0; $i < $1; i++ ))
do
echo -n '* '
done
echo ''
for(( i=1; $i < $1-1; i++ ))
do
    for(( j=0; $j < $i; j++ ))
    do
        echo -n " "
    done
    echo -n "*"
    for(( j=2; $j < $1-$i; j++ ))
    do
    echo -n '  '
    done
    echo ' *'
done

for(( i=0; $i < $1-1; i++ ))
do
    echo -n ' '
done
echo '*'

for(( i = $1-1; $i > 1; i-- ))
do
    for(( j=1; $j < $i; j++ ))
    do
        echo -n " "
    done
    echo -n "*"
    for(( j=1; $j < $1-$i; j++ ))
    do
    echo -n '  '
    done
    echo ' *'
done
for(( i=0; $i < $1; i++ ))
do
echo -n '* '
done
echo ''
return
fi
echo "Ошибка: параметр некоректный или отсутствует"
